let obj = { a: { x: [1, 2, 3, 4, 5, { m: [1, 2, 3, 4, 5, 8] }] } };
function sumofobj(obj) {
  let sum = 0;
  if (Array.isArray(obj)) {
    for (let i = 0; i < obj.length; i++) {
      if (typeof obj[i] === "number") {
        sum += obj[i];
      } else {
        sum += sumofobj(obj[i]);
      }
    }
  } else {
    let keys = Object.keys(obj);
    for (let i = 0; i < keys.length; i++) {
      sum += sumofobj(obj[keys[i]]);
    }
  }
  return sum;
}
